## code

```python
import itertools
from collections import deque

#입력
N, M = map(int,input().split())
lab = [list(map(int,input().split())) for _ in range(N)]

mn = 9999999999 #최솟값
virus = [] #바이러스 위치 담을 리스트
cnt = 0 

for i in range(N):
    for j in range(N):
        if lab[i][j] == 2:
            virus.append((i,j)) #바이러스 위치 담기
        elif lab[i][j] == 1: 
            cnt += 1 #벽 수

#바이러스 위치 조합 담기
tmp_virus = list(itertools.combinations(virus,M))

#위치 
for tmp in tmp_virus:
    q = deque(tmp)
    visited = [[-1]*N for _ in range(N)]

    space = N * N - cnt - M #바이러스를 퍼트려야 할 곳 수
    
    #BFS
    for x,y in q:
        visited[x][y] = 0
	
    while q:
        tx, ty = q.popleft()

        for dx, dy in ((-1,0),(0,1),(1,0),(0,-1)):
            nx = tx + dx
            ny = ty + dy
            # 방문 안했고, 범위 안 있고, 벽이 아닐 경우
            if 0 <= nx < N and 0 <= ny < N and lab[nx][ny] != 1 and visited[nx][ny] == -1:
                visited[nx][ny] = visited[tx][ty] + 1
                space -= 1 
                q.append((nx,ny))

    if space == 0: #모두 퍼트렸을 때
        time = visited[tx][ty]
        if mn > time:
            mn = time


if mn == 9999999999:
    print(-1)
else:
    print(mn)

```
