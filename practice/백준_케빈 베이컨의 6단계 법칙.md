# 케빈 베이컨의 6단계 법칙 (1389)



## code

```python
from collections import deque

N, M = map(int,input().split())
graph = [[] for _ in range(N+1)]

for _ in range(M):
    A, B = map(int,input().split())
    graph[A].append(B)
    graph[B].append(A)

mn = 99999

for i in range(1,N+1):
	# BFS
    q = deque()
    q.append(i)
    visited = [0] * (N + 1)
    visited[i] = 1
    res= 0

    while q:
        tmp = q.popleft()
        for j in graph[tmp]:
            if not visited[j]:
                q.append(j)
                visited[j] = visited[tmp] + 1
                res += visited[j]

    if mn > res:
        mn = res
        idx = i

print(idx)
```
