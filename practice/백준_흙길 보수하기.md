# 흙길 보수하기 (1911)

```python
N, L = map(int,input().split())
hole = [list(map(int,input().split())) for _ in range(N)]
hole.sort()

cnt = 0 
now = 0 # 현재 널빤지 끝 위치

for x, y in hole:

    tmp = 0
    if now > y: # 널빤지가 깔렸으면
        continue # 다음
    if now < x: # 웅덩이의 시작부터 깔려야 널빤지를 최소로 할 수 있음
        now = x

    tmp = (y - now) // L # 몫
    if (y - now) % L: # 나머지 있으면
        cnt += tmp + 1 # 몫+1 깔고
        now += (tmp + 1) * L # 위치 갱신
    else:
        cnt += tmp # 없으면 몫만
        now += tmp * L # 갱신

print(cnt)
```
