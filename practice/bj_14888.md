# 연산자 끼워넣기



## code

```python
from itertools import permutations

N = int(input())
nums = list(map(int,input().split()))
operator = list(map(int,input().split())) # +, -, x, //
op = []

for i in range(4):
    for _ in range(operator[i]):
        op.append(i)

op_per = list(permutations(op, N-1))
result = []

for i in op_per:
    res = nums[0]
    for j in range(N-1):
        if i[j] == 0:
            res = res + nums[j+1]
        elif i[j] == 1:
            res = res - nums[j+1]
        elif i[j] == 2:
            res = res * nums[j+1]
        elif i[j] == 3:
            if res >= 0:
                res = res // nums[j+1]
            else:
                res = -((-res)//nums[j+1])

    result.append(res)

print(max(result))
print(min(result))

```
